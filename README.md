# *installation*
--- 

# Use the package manager pip to install foobar


pip install foobar
 

# *Usage*
--- 


import foobar
foobar.pluralize('word') #returns words
foobar.pluralize('goose') #returns geese
foobar.pluralize('phenomena') #returns 'phenomenon'_
